1#include <vector>
2using namespace std;
3
4class Solution {
5public:
6    int scoreDifference(vector<int>& nums) {
7        int first = 0, second = 0;
8        bool activeFirst = true;
9        for (int i = 0; i < (int)nums.size(); ++i) {
10            if (nums[i] % 2 == 1) activeFirst = !activeFirst;
11            if (i % 6 == 5) activeFirst = !activeFirst;
12            if (activeFirst) first += nums[i];
13            else second += nums[i];
14        }
15        return first - second;
16    }
17};
