1class Solution {
2public:
3    vector<int> twoSum(vector<int>& nums, int target) {
4        unordered_map<int,int> mpp;
5        for(int i=0;i<nums.size();i++){
6            int num=nums[i];
7            int diff=target-num;
8            if(mpp.find(diff)!=mpp.end()){

12        }
13        return {-1,-1};
14    }
